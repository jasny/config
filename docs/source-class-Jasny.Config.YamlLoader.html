53:     {

55:             case 'yaml': $data = yaml_parse_file($file); break;
56:             case 'syck': $data = syck_load(file_get_contents($file)); break;
57:             case 'spyc': $data = \Spyc::YAMLLoad($file); break;
58:             default: return null;
